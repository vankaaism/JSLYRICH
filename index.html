function sendQrEmails() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = sheet.getDataRange().getValues(); var draft = getEmailDraft(); if (!draft) { Logger.log("No email draft found."); return; } for (var i = 1; i < data.length; i++) { var qrcodenumber = data[i][2]; var participantCode = data[i][2]; var email = data[i][3]; var name = data[i][1]; if (!email) continue; var qrCodeUrl = `https://api.qrserver.com/v1/create-qr-code/?size=150x150&data=${encodeURIComponent(qrcodenumber)}`; var qrCodeBlob; try { var response = UrlFetchApp.fetch(qrCodeUrl); qrCodeBlob = response.getBlob(); } catch (e) { Logger.log("Failed to fetch QR code for " + email + ": " + e.toString()); continue; } var emailBody = draft.replace("{{name}}", name) .replace("{{type}}", participantCode) .replace("{{qr}}", '') MailApp.sendEmail({ to: email, subject: "See You at TEDxTU Pre-Event 1 â€“ Your Ticket is Here" + name, htmlBody: emailBody, inlineImages: { qrCode: qrCodeBlob, } }); } Logger.log("Emails sent successfully!"); } function getEmailDraft() { var drafts = GmailApp.getDrafts(); for (var i = 0; i < drafts.length; i++) { var draft = drafts[i].getMessage(); return draft.getBody(); // Return first draft found } return null; }
